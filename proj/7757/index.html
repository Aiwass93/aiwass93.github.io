\( \displaystyle \begin{aligned} &t \in \mathbb Z_{\ge 0}\ \text{ms}, \qquad p_j=\text{the }j\text{-th prime} \\[0.6em] d(t) &= \begin{cases} 0 & 0 \le t < 10^9 \\ \lfloor \log_{10} t \rfloor & \text{otherwise} \end{cases} \\ \chi(t) &= \max\!\left(0,1-\frac{d(t)}{7757}\right) \\ \alpha(t) &= \sum_{j=1}^{512}\frac{t \bmod p_j}{p_j}, \qquad \beta(t) = \sum_{j=513}^{1024}\frac{t \bmod p_j}{p_j} \\[0.6em] f_c(t) &= \max\!\left(20,100+\chi(t)\left(500+300\sin\alpha(t)\right)\right) \\ f_m(t) &= f_c(t)\left(2+0.71828\chi(t)\right) \\ I(t) &= 1+\chi(t)\left(19+10\sin\beta(t)\right) \\ \eta(t) &= 0.5\chi(t), \qquad \rho(t)=0.9\chi(t), \qquad g(t)=1+99\chi(t) \\[0.6em] \tau_n &= t_0+\left\lfloor w_{k(n)}-w_0\right\rfloor, \qquad U[n]=\operatorname{Math.random}()\in[0,1), \qquad N[n]=2U[n]-1 \\ \phi_c[0] &= \phi_m[0]=0, \qquad y[-1]=0 \\ x[n] &= \sin\!\left(\phi_c[n]+I(\tau_n)\sin\phi_m[n]\right) \\ y[n] &= \tanh\!\left(g(\tau_n)\left((1-\eta(\tau_n))x[n]+\eta(\tau_n)N[n]+\rho(\tau_n)y[n-1]\right)\right) \\ \phi_c[n+1] &= \left(\phi_c[n]+2\pi\frac{f_c(\tau_n)}{R_s}\right) \bmod 2\pi \\ \phi_m[n+1] &= \left(\phi_m[n]+2\pi\frac{f_m(\tau_n)}{R_s}\right) \bmod 2\pi \\ (L[n],R[n]) &= 0.2\,(y[n],-y[n]) \end{aligned} \)
